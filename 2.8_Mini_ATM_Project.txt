# ATM Mini Project with PIN

# Preset correct PIN and initial balance
correct_pin = 1234
balance = 10000

print("Welcome to SBI Bank")

# Ask user to enter PIN
pin = int(input("Enter your PIN: "))

# Check if PIN is correct
if pin != correct_pin:
    print("Incorrect PIN! Access denied.")
else:
    print("Login successful!\n")
    
    # Loop to display ATM menu
    while True:
        print("ATM Menu")
        print("1. Check Balance")
        print("2. Deposit")
        print("3. Withdraw")
        print("4. Exit")
        
        # Take user choice
        choice = int(input("Enter your choice (1-4): "))
        
        if choice == 1:
            # Display current balance
            print("Your balance is:", balance)
        
        elif choice == 2:
            # Deposit money
            amount = int(input("Enter amount to deposit: "))
            balance += amount
            print(amount, "deposited. New balance:", balance)
        
        elif choice == 3:
            # Withdraw money
            amount = int(input("Enter amount to withdraw: "))
            if amount > balance:
                print("Insufficient balance!")
            else:
                balance -= amount
                print(amount, "withdrawn. New balance:", balance)
        
        elif choice == 4:
            # Exit ATM
            print("Thank you for using SBI ATM!")
            break
        
        else:
            # Invalid choice
            print("Invalid choice! Please try again.")
