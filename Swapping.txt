# Swap Two Numbers

# Method 1: Using temporary variable
a = int(input("Enter first number: "))
b = int(input("Enter second number: "))

print("Before swap: a =", a, "b =", b)

temp = a
a = b
b = temp

print("After swap (using temp variable): a =", a, "b =", b)

# Method 2: Without using temporary variable
a = int(input("\nEnter first number again: "))
b = int(input("Enter second number again: "))

print("Before swap: a =", a, "b =", b)

a = a + b
b = a - b
a = a - b

print("After swap (without temp variable): a =", a, "b =", b)

# Method 3: Using Python multiple assignment
a = int(input("\nEnter first number again: "))
b = int(input("Enter second number again: "))

print("Before swap: a =", a, "b =", b)

a, b = b, a

print("After swap (using multiple assignment): a =", a, "b =", b)
